\newcommand\lineEntry[2]
{
	\bf #1	&	#2 \\
}

\newcommand\inventory[1]
{
	\begingroup
		\setkeys{inventory}{name = none, class = none, acquired = none, hasAcquired = 0, description = none, trueDescription = none, hasTrueDescription = 0, purchaseCost = 0, hasCost = 0, sellValue = 0,amount = 0}
	
		\setkeys{inventory}{#1}
		
		
		\subsubsection{\name}
		
		\begin{center}
			\begin{tabular}{r p{\w cm}}
			\if\hasAcquired1
				\lineEntry{Acquired:}{\acquired}
			\fi
			\if\hasCost1
				\lineEntry{Cost:}{\purchaseCost}
			\fi
			\lineEntry{Amount:}{\amount}
			\end{tabular}
		\end{center}
		
		\description
		
	\endgroup

}


\newcommand\GMinventory[1]
{
	\begingroup

		\setkeys{inventory}{name = none, class = none, acquired = none, hasAcquired = 0, description = none, trueDescription = none, hasTrueDescription = 0, purchaseCost = 0, hasCost = 0, sellValue = 0,amount = 0,isOwned = 0, owners = None}
	
		\setkeys{inventory}{#1}
		
		
		\subsubsection{\name}
		
		\begin{center}
			\begin{tabular}{r p{\w cm}}
			\if\hasAcquired1
				\lineEntry{Acquired:}{\acquired}
			\fi
			\if\hasCost1
				\lineEntry{Cost:}{\purchaseCost}
			\fi
			\lineEntry{Sell Value:}{\sellValue}
			\if\isOwned1
				\lineEntry{Owners:}{\parbox[t]{4cm}{\owners}}
			\fi
			\if\isOwned0
				\lineEntry{Owners:}{No-one currently owns this item}
			\fi
				
			\end{tabular}
		\end{center}
		
		\description
		
		\if\hasTrueDescription1
		~
		
			\textbf{Hidden Description}
			
			\trueDescription
		\fi
	\endgroup

}
